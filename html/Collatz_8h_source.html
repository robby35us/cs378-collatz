00055 int _getCycLen(long int num) {
00056    assert(num > 0);
00057    if(num < cacheLen && _cache[num] > 0)
00058       return _cache[num];
00059    long int index = num;
00060    int cycLen = 1;
00061    while(index > 1) {
00062       if(index < cacheLen && _cache[index] > 0) {
00063          cycLen += _cache[index] - 1;
00064          break;}
00065       else if(index % 2 == 0)
00066          index = index / 2;
00067       else
00068          index = 3 * index + 1;
00069       ++cycLen;
00070       assert(index > 0);}
00071    if(num < cacheLen && _cache[num] == 0)
00072       _cache[num] = cycLen;
00073    assert(num > 0);
00074    return cycLen;}
